I often use R, Python, SAS, SQL, and Bash in different tasks at the same time. To make the switch between different languages easier, I built a list of the commonly-used script for a quick reference.

A simulated sample data set with missing values and duplicated values was used. The reticulate package was used to call R data from Phthon.

set.seed(1)
N <- 1000
df <- tibble(
dimension1 = sample(c("I", "II", "III"), N, replace = T),
dimension2 = sample(c("A", "B", "C"), N, replace = T),
measure1 = sample(1:10, N, replace = T),
measure2 = sample(1:10, N, replace = T)
)

df <- as_tibble(lapply(df, function(r) r[sample(c(TRUE, NA), prob = c(0.85, 0.15), size = length(r), replace = TRUE) ]))

Missing Values

R

# count missing by columns
colSums(is.na(df)) 
## dimension1 dimension2   measure1   measure2 
##        173        153        169        147
# drop rows with missing
no_missing <- na.omit(df) 

# drop rows with all missing
no_all_row_missing <- 
  df[rowSums(!is.na(df)) > 1, ] 

# drop columns with all missing
no_all_col_missing <- 
  df[colSums(!is.na(df)) > 0]
# fill all missing by 0
fill_missing <- df %>%
  replace(., is.na(.), 0)

# fill missing by columns
fill_col_missing <- df %>%
  replace_na(list(dimension1 = "Unknown", measure1 = 0))

Python

# count missing by columns
print(df.isnull().sum()) 
# drop any missing
no_missing = df.dropna() 
# drop rows with all missing
no_all_row_missing = df.dropna(how="all") 
# drop columns with all missing
no_all_col_missing = df.dropna(axis=1, how='all') 
# fill all missing by 0
fill_missing = df.fillna(0) 
# fill missing by columns
fillValues = {"dimension1": "Unknown", "measure1": 0} 
fill_col_missing = df.fillna(value=fillValues)
